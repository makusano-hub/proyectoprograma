#include "enemigo.h"

int colision(jugador *j,enemigo *e){
    int anchoJ = al_get_bitmap_width(j->sprite); 
    int altoJ = al_get_bitmap_height(j->sprite);
    
    int anchoE = al_get_bitmap_width(j->sprite);
    int altoJ = al_get_bitmap_height(j->sprite);

    return(j->ejex < e->posx + anchoE && j->ejex + anchoJ > e->posx);
}
